# 개발 환경 설정

```bash
# git lfs 설치 이후 clone
## Linux + Docker
git lfs install && git clone --recurse-submodules https://github.com/pray92/reachy-mini-jisu.git

## MacOS
brew install git-lfs
git lfs install 

## 이후 공통
git submodule update --init --recursive
git submodule update --remote

# reachy-mini 설치를 위해선 python 3.10 설치 필수, 해당 패키지는 numpy2.2.5 필요하며 이는 Python 3.10 버전부터 사용 가능
## conda 기반 설치
conda create -n orokapangyo_reach_mini python=3.10
conda activate orokapangyo_reach_mini

## Linux + Docker 기반
### PyGObject 빌드에 시스템 라이브러리 필요
apt-get install -y libcairo2-dev libgirepository1.0-dev pkg-config gir1.2-glib-2.0
### GStreamer 라이브러리 설치
apt-get install -y gstreamer1.0-tools gstreamer1.0-plugins-base gstreamer1.0-plugins-good gir1.2-gstreamer-1.0 gir1.2-gst-plugins-base-1.0
### GstApp 네임스페이스 추가
apt-get install -y gstreamer1.0-plugins-bad gir1.2-gst-plugins-bad-1.0 libgstrtspserver-1.0-dev

## MacOS 기반 (스킵)

## 이후 공통
### Reachy Mini SDK 설치
pip install reachy-mini
pip install "reachy-mini[mujoco]"

## 시뮬레이션 환경 실행
reachy-mini-daemon --sim --scene minimal --deactivate-audio

### 3D 작동이 되지 않는 관계로 mjpython 기반 실행
mjpython $(which reachy-mini-daemon) --sim --scene minimal --deactivate-audio

### Docker 환경 : GPU/디스플레이가 없는 관계로 headless 모드 실행
reachy-mini-daemon --sim --scene minimal --headless
```
