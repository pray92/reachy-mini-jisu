# Audio

MuJoCo 시뮬레이션 환경에서 Reachy Mini와 상호작용하기 위한 음성 처리 시스템은 크게 **입력(마이크), 처리(인식), 출력(스피커)** 세 단계로 나뉩니다. MuJoCo 자체는 물리 엔진이므로 음성 처리는 파이썬의 표준 라이브러리들을 활용하여 구현합니다.

가장 대중적이고 가벼운 라이브러리인 `sounddevice`(입출력)와 `SpeechRecognition`(음성 인식)을 사용하는 방법을 정리해 드립니다.

```python
# 방법 : 키워드 감지 개선 버전 (여러 인사 표현 인식)
from reachy_mini import ReachyMini
import time
import numpy as np
import tempfile
import os
from scipy.io.wavfile import write

try:
    import speech_recognition as sr
    SPEECH_RECOGNITION_AVAILABLE = True
except ImportError:
    SPEECH_RECOGNITION_AVAILABLE = False

def enhanced_greet_action(reachy):
    """
    개선된 인사 동작 (더 자연스러운 움직임)
    """
    from reachy_mini.utils import create_head_pose
    
    print("👋 안녕하세요!")
    
    # 1. 고개를 살짝 숙이며 인사
    reachy.goto_target(
        head=create_head_pose(pitch=-12, degrees=True),
        antennas=[0.2, 0.2],
        body_yaw=np.deg2rad(-5),
        duration=0.6
    )
    time.sleep(0.4)
    
    # 2. 고개를 들며 미소
    reachy.goto_target(
        head=create_head_pose(pitch=8, yaw=3, degrees=True),
        antennas=[0.6, 0.6],
        body_yaw=0,
        duration=0.5
    )
    time.sleep(0.3)
    
    # 3. 좌우로 살짝 흔들며 인사 (2번)
    for i in range(2):
        reachy.goto_target(
            head=create_head_pose(yaw=-12, pitch=3, degrees=True),
            antennas=[0.3, 0.7],
            duration=0.25
        )
        time.sleep(0.15)
        reachy.goto_target(
            head=create_head_pose(yaw=12, pitch=3, degrees=True),
            antennas=[0.7, 0.3],
            duration=0.25
        )
        time.sleep(0.15)
    
    # 4. 초기 위치로 복귀
    reachy.goto_target(
        head=create_head_pose(),
        antennas=[0.0, 0.0],
        body_yaw=0,
        duration=0.6
    )
    
    print("✅ 인사 완료!")

def keyword_detection_greet(reachy, duration=60, language='ko-KR'):
    """
    여러 키워드를 감지하여 인사하는 함수
    
    인식 키워드:
    - 한국어: "인사해줘", "인사", "안녕", "안녕하세요", "헬로"
    - 영어: "hello", "hi", "greet", "greeting"
    """
    if not SPEECH_RECOGNITION_AVAILABLE:
        print("❌ speech_recognition 패키지가 필요합니다.")
        return
    
    recognizer = sr.Recognizer()
    
    # 인사 키워드 목록
    greet_keywords_ko = ["인사", "안녕", "헬로", "hello"]
    greet_keywords_en = ["hello", "hi", "greet", "greeting"]
    
    if language == 'ko-KR':
        keywords = greet_keywords_ko
    else:
        keywords = greet_keywords_en
    
    print(f"=== 키워드 감지 모드 (지속 시간: {duration}초) ===")
    print(f"인식 키워드: {', '.join(keywords)}")
    print("'Ctrl+C'로 중단할 수 있습니다.")
    
    reachy.media.start_recording()
    
    try:
        start_time = time.time()
        last_check_time = time.time()
        check_interval = 2.0
        last_action_time = 0
        action_cooldown = 3.0
        
        while time.time() - start_time < duration:
            if time.time() - last_check_time > check_interval:
                # 최근 2초간의 오디오 수집
                audio_samples = []
                collect_start = time.time()
                
                while time.time() - collect_start < 2.0:
                    sample = reachy.media.get_audio_sample()
                    if sample is not None:
                        audio_samples.append(sample)
                    time.sleep(0.01)
                
                if audio_samples and (time.time() - last_action_time) > action_cooldown:
                    # 오디오 데이터 처리
                    combined = np.concatenate(audio_samples)
                    if combined.ndim == 2:
                        if combined.shape[0] < combined.shape[1]:
                            combined = np.mean(combined, axis=0)
                        else:
                            combined = np.mean(combined, axis=1)
                    
                    max_val = np.max(np.abs(combined))
                    if max_val > 0:
                        combined = combined / max_val
                    
                    audio_int16 = (combined * 32767).astype(np.int16)
                    sample_rate = reachy.media.get_input_audio_samplerate()
                    
                    # 임시 파일로 저장
                    with tempfile.NamedTemporaryFile(delete=False, suffix=".wav") as tmp_wav:
                        write(tmp_wav.name, sample_rate, audio_int16)
                        
                        try:
                            with sr.AudioFile(tmp_wav.name) as source:
                                audio_data = recognizer.record(source)
                                try:
                                    text = recognizer.recognize_google(audio_data, language=language)
                                    text_lower = text.lower()
                                    print(f"\n🎤 인식된 말: '{text}'")
                                    
                                    # 키워드 확인
                                    found_keyword = False
                                    for keyword in keywords:
                                        if keyword.lower() in text_lower:
                                            found_keyword = True
                                            break
                                    
                                    if found_keyword:
                                        print("✅ 인사 키워드 감지!")
                                        enhanced_greet_action(reachy)
                                        last_action_time = time.time()
                                        time.sleep(1.0)
                                    else:
                                        print(f"   (인사 키워드가 아닙니다)")
                                        
                                except sr.UnknownValueError:
                                    pass  # 음성 인식 실패는 무시
                                except sr.RequestError as e:
                                    print(f"\r음성 인식 서비스 오류: {e}", end="", flush=True)
                        finally:
                            if os.path.exists(tmp_wav.name):
                                os.remove(tmp_wav.name)
                
                last_check_time = time.time()
            
            time.sleep(0.1)
        
        print("\n키워드 감지 모드 종료")
    
    except KeyboardInterrupt:
        print("\n사용자가 중단했습니다.")
    finally:
        reachy.media.stop_recording()

# 사용 예시
if SPEECH_RECOGNITION_AVAILABLE:
    with ReachyMini() as reachy:
        keyword_detection_greet(reachy, duration=60, language='ko-KR')
else:
    print("speech_recognition 패키지를 먼저 설치하세요.")
```

**Reachy Mini 로봇이 마이크를 통해 실시간으로 음성을 듣고, 특정 키워드(인사 등)가 감지되면 정해진 제스처를 수행**하도록 설계

---

### 1. 동작 제어부 (`enhanced_greet_action`)

이 함수는 로봇이 단순히 팔을 흔드는 것을 넘어, **머리, 안테나, 몸체(Yaw)**를 동시에 사용하여 감정을 표현하는 자연스러운 인사를 구현합니다.

- **다관절 협응 제어**: `head` (피치/요), `antennas` (좌우 독립 제어), `body_yaw`를 한 번의 `goto_target` 명령으로 동시에 움직입니다.
- **시퀀스 구성**:
    1. **숙이기**: 고개를 -12도 숙여 예의 바른 인사 표현.
    2. **미소/반가움**: 고개를 들고 안테나를 높게(0.6) 세워 긍정적인 상태 표현.
    3. **도리도리**: 고개를 좌우로 흔들며 안테나를 비대칭으로 움직여 귀여운 느낌 전달.
    4. **복귀**: 모든 관절을 초기 상태(`0, 0`)로 부드럽게 복구.

---

### 2. 음성 처리 및 인식부 (`keyword_detection_greet`)

이 부분은 하드웨어로부터 들어오는 날것(Raw)의 오디오 데이터를 AI 모델이 이해할 수 있는 형태로 가공하는 과정입니다.

- **실시간 버퍼링**: `reachy.media.get_audio_sample()`을 통해 짧은 간격(2초)으로 오디오 데이터를 수집합니다.
- **신호 처리 (Signal Processing)**:
    - **모노 변환**: 다채널 오디오를 `np.mean`을 통해 단일 채널(Mono)로 합칩니다. (인식률 향상)
    - **정규화 (Normalization)**: 소리가 너무 작거나 클 경우를 대비해 최대 진폭으로 나누어 일정한 크기를 유지합니다.
    - **형식 변환**: 부동 소수점(Float) 데이터를 16비트 정수(Int16) 형식으로 변환하여 WAV 표준에 맞춥니다.
- **인식 로직**:
    - `Google Web Speech API`를 사용하여 음성을 텍스트(STT)로 바꿉니다.
    - **키워드 매칭**: 인식된 문장에 "인사", "안녕", "hello" 등이 포함되어 있는지 검사합니다.
    - **쿨다운(Cooldown)**: `action_cooldown`을 두어, 한 번 인사한 뒤 3초 동안은 중복 반응하지 않도록 설계되었습니다.

---

### 3. 코드의 기술적 특징 및 장단점

### **👍 장점 (Strengths)**

- **자원 관리**: `tempfile.NamedTemporaryFile`을 사용하여 음성 파일이 하드디스크에 쌓이지 않고 자동으로 삭제되도록 관리했습니다.
- **유연성**: 한국어와 영어를 모두 지원하며, 키워드 리스트를 쉽게 확장할 수 있습니다.
- **안정성**: `try-except` 구문을 통해 음성 인식 실패나 네트워크 오류 시에도 프로그램이 중단되지 않고 다음 시도를 이어갑니다.
